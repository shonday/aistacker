"use client"

import { useState, useCallback } from "react"
import { Button, Spinner } from "@/components/ui"
import { Check } from "lucide-react"

const japaneseRegex = /^[\u3000-\u303f\u3040-\u309f\u30a0-\u30ff\uff00-\uff9f\u4e00-\u9faf\u3400-\u4dbf]+$/;
const kanjiRegex = /[\u4e00-\u9faf\u3400-\u4dbf]/g;
const hiraganaRegex = /[\u3040-\u309f]/g;
const katakanaRegex = /[\u30a0-\u30ff]/g;

export default function NinjaTextCount() {
  const [text, setText] = useState("")
  const [error, setError] = useState("")
  const [counts, setCounts] = useState({ kanji: 0, hiragana: 0, katakana: 0, total: 0 })
  const [loading, setLoading] = useState(false)
  const [copied, setCopied] = useState<string | null>(null)

  const handleTextChange = useCallback((e: React.ChangeEvent<HTMLTextAreaElement>) => {
    const newText = e.target.value
    setText(newText)
    if (!japaneseRegex.test(newText)) {
      setError("Invalid Japanese text")
    } else {
      setError("")
    }
    countCharacters(newText)
  }, [])

  const countCharacters = useCallback((text: string) => {
    const kanjiCount = (text.match(kanjiRegex) || []).length
    const hiraganaCount = (text.match(hiraganaRegex) || []).length
    const katakanaCount = (text.match(katakanaRegex) || []).length
    const totalCount = text.length
    setCounts({ kanji: kanjiCount, hiragana: hiraganaCount, katakana: katakanaCount, total: totalCount })
  }, [])

  const copyToClipboard = useCallback(async (text: string, id: string) => {
    try {
      await navigator.clipboard.writeText(text)
    } catch {
      const el = document.createElement("textarea")
      el.value = text
      document.body.appendChild(el)
      el.select()
      document.execCommand("copy")
      document.body.removeChild(el)
    }
    setCopied(id)
    setTimeout(() => setCopied(null), 1500)
  }, [])

  const handleExport = useCallback((format: string) => {
    if (format === "csv") {
      const csvText = `Kanji,${counts.kanji}\nHiragana,${counts.hiragana}\nKatakana,${counts.katakana}\nTotal,${counts.total}`
      copyToClipboard(csvText, "export")
    } else if (format === "json") {
      const jsonText = JSON.stringify(counts, null, 2)
      copyToClipboard(jsonText, "export")
    }
  }, [counts, copyToClipboard])

  const handleKeyDown = useCallback((e: React.KeyboardEvent) => {
    if (e.ctrlKey && e.key === "Enter" || e.metaKey && e.key === "Enter") {
      handleExport("csv")
    }
  }, [handleExport])

  return (
    <div className="bg-background p-4 rounded">
      <h2 className="text-foreground mb-4">Ninja Text Counter</h2>
      {error ? (
        <p className="text-destructive mb-4">{error}</p>
      ) : (
        <p className="text-muted-foreground mb-4">Enter Japanese text to count characters</p>
      )}
      <textarea
        className="bg-muted p-2 rounded w-full h-40"
        value={text}
        onChange={handleTextChange}
        onKeyDown={handleKeyDown}
        placeholder="Enter Japanese text"
      />
      {loading ? (
        <Spinner className="mx-auto mt-4" />
      ) : (
        <div className="mt-4">
          <p className="text-foreground mb-2">Kanji: {counts.kanji}</p>
          <p className="text-foreground mb-2">Hiragana: {counts.hiragana}</p>
          <p className="text-foreground mb-2">Katakana: {counts.katakana}</p>
          <p className="text-foreground mb-2">Total: {counts.total}</p>
          <Button
            className="mb-2"
            onClick={() => handleExport("csv")}
          >
            Export as CSV
          </Button>
          <Button
            className="mb-2"
            onClick={() => handleExport("json")}
          >
            Export as JSON
          </Button>
          <button
            className="bg-primary text-primary-foreground p-2 rounded"
            onClick={() => copyToClipboard(JSON.stringify(counts, null, 2), "result")}
          >
            {copied === "result" ? "Copied ✓" : "Copy result"}
          </button>
        </div>
      )}
    </div>
  )
}
